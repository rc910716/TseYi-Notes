	.globl SYMBOL_NAME(name); \
	LENTRY(name)
	
#define SYMBOL_NAME(X)		X

#define SYMBOL_NAME_LABEL(X)	X##:

#define LENTRY(name) \
	ALIGN; \
	SYMBOL_NAME_LABEL(name)
	
#ifndef __ALIGN
#define __ALIGN .align		4


.globl save_boot_params;
.align		4; save_boot_params

	
	
	
	


ENDPROC(save_boot_params)

#define ENDPROC(name) \
	.type name STT_FUNC; \
	END(name)
	
#ifndef END
#define END(name) \
	.size name, .-name
#endif
	

#伪指令，需要连接器声明这是一个函数
